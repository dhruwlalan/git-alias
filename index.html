$ git branchAll

$ git co <branch>

$ git pullRb <remote-branch>

delete remote branch

$ git deleteRb <remote-branch>

$ git deleteLbRb <branch>

$ git listDeletedRb

$ git track <remote-branch>

$ git deleteLbRbTdr <remote> <branch>
